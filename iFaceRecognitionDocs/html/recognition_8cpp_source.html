122 // Generate an approximately reconstructed face by back-projecting the eigenvectors & eigenvalues of the given (preprocessed) face.
123 Mat reconstructFace(const Ptr<FaceRecognizer> model, const Mat preprocessedFace)
124 {
125  // Since we can only reconstruct the face for some types of FaceRecognizer models (ie: Eigenfaces or Fisherfaces),
126  // we should surround the OpenCV calls by a try/catch block so we don't crash for other models.
127  try {
128 
129  // Get some required data from the FaceRecognizer model.
130  Mat eigenvectors = model->get<Mat>("eigenvectors");
131  Mat averageFaceRow = model->get<Mat>("mean");
132 
133  int faceHeight = preprocessedFace.rows;
134 
135  // Project the input image onto the PCA subspace.
136  Mat projection = subspaceProject(eigenvectors, averageFaceRow, preprocessedFace.reshape(1,1));
137  //printMatInfo(projection, "projection");
138 
139  // Generate the reconstructed face back from the PCA subspace.
140  Mat reconstructionRow = subspaceReconstruct(eigenvectors, averageFaceRow, projection);
141  //printMatInfo(reconstructionRow, "reconstructionRow");
142 
143  // Convert the float row matrix to a regular 8-bit image. Note that we
144  // shouldn't use "getImageFrom1DFloatMat()" because we don't want to normalize
145  // the data since it is already at the perfect scale.
146 
147  // Make it a rectangular shaped image instead of a single row.
148  Mat reconstructionMat = reconstructionRow.reshape(1, faceHeight);
149  // Convert the floating-point pixels to regular 8-bit uchar pixels.
150  Mat reconstructedFace = Mat(reconstructionMat.size(), CV_8U);
151  reconstructionMat.convertTo(reconstructedFace, CV_8U, 1, 0);
152  //printMatInfo(reconstructedFace, "reconstructedFace");
153 
154  return reconstructedFace;
155 
156  } catch (cv::Exception e) {
157  //cout << "WARNING: Missing FaceRecognizer properties." << endl;
158  return Mat();
159  }
160 }

163 // Compare two images by getting the L2 error (square-root of sum of squared error).
164 double getSimilarity(const Mat A, const Mat B)
165 {
166  if (A.rows > 0 && A.rows == B.rows && A.cols > 0 && A.cols == B.cols) {
167  // Calculate the L2 relative error between the 2 images.
168  double errorL2 = norm(A, B, CV_L2);
169  // Convert to a reasonable scale, since L2 error is summed across all pixels of the image.
170  double similarity = errorL2 / (double)(A.rows * A.cols);
171  return similarity;
172  }
173  else {

175  return 100000000.0; // Return a bad value
